// ===================== 讯飞鉴权 =====================
async function getXFUrl() {
  const host = 'iat-api.xfyun.cn';
  const date = new Date().toUTCString();
  const requestLine = 'GET /v2/iat HTTP/1.1';
  const signOrigin = `host: ${host}\ndate: ${date}\n${requestLine}`;
  const enc = new TextEncoder();
  const key = await crypto.subtle.importKey(
    'raw', enc.encode(XF_API_SECRET),
    { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']
  );
  const sig    = await crypto.subtle.sign('HMAC', key, enc.encode(signOrigin));
  const sigB64 = btoa(String.fromCharCode(...new Uint8Array(sig)));
  const authOrigin = `api_key="${XF_API_KEY}", algorithm="hmac-sha256", headers="host date request-line", signature="${sigB64}"`;
  const auth = btoa(authOrigin);
  return `wss://${host}/v2/iat?authorization=${encodeURIComponent(auth)}&date=${encodeURIComponent(date)}&host=${host}`;
}

// ===================== PCM转换 =====================
// 用回调式 decodeAudioData，兼容所有浏览器
function convertToPCM(blob) {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();

    reader.onerror = () => reject(new Error('读取音频文件失败'));

    reader.onload = (e) => {
      const arrayBuffer = e.target.result;
      let ctx;
      try {
        ctx = new (window.AudioContext || window.webkitAudioContext)({ sampleRate: 16000 });
      } catch(err) {
        return reject(new Error('AudioContext 初始化失败：' + err.message));
      }

      // 超时保护：8秒解码还没完就报错
      const decodeTimeout = setTimeout(() => {
        try { ctx.close(); } catch(_) {}
        reject(new Error('音频解码超时'));
      }, 8000);

      // 用回调形式（而非Promise形式），兼容性更好
      ctx.decodeAudioData(
        arrayBuffer,
        (audioBuffer) => {
          clearTimeout(decodeTimeout);
          try {
            const pcm   = audioBuffer.getChannelData(0);
            const int16 = new Int16Array(pcm.length);
            for (let i = 0; i < pcm.length; i++) {
              int16[i] = Math.max(-32768, Math.min(32767, Math.round(pcm[i] * 32767)));
            }
            ctx.close().catch(() => {});
            resolve(new Blob([int16.buffer], { type: 'audio/L16' }));
          } catch(err) {
            reject(new Error('PCM转换失败：' + err.message));
          }
        },
        (err) => {
          clearTimeout(decodeTimeout);
          try { ctx.close(); } catch(_) {}
          // 解码失败时直接用原始blob发送，讯飞服务端也能处理webm
          console.warn('decodeAudioData失败，将使用原始音频：', err);
          resolve(blob);
        }
      );
    };

    reader.readAsArrayBuffer(blob);
  });
}

// ===================== 讯飞识别 =====================
function recognizeAudio(audioBlob) {
  return new Promise(async (resolve, reject) => {
    let url;
    try {
      url = await getXFUrl();
    } catch(err) {
      return reject(new Error('鉴权失败：' + err.message));
    }

    let pcmBlob;
    try {
      pcmBlob = await convertToPCM(audioBlob);
    } catch(err) {
      return reject(new Error('音频处理失败：' + err.message));
    }

    let pcmArray;
    try {
      pcmArray = new Uint8Array(await pcmBlob.arrayBuffer());
    } catch(err) {
      return reject(new Error('音频读取失败：' + err.message));
    }

    const segments = [];   // 每个分段的识别结果
    let settled    = false;

    const done = (text) => {
      if (settled) return;
      settled = true;
      resolve(text);
    };

    const fail = (err) => {
      if (settled) return;
      settled = true;
      reject(err);
    };

    let ws;
    try {
      ws = new WebSocket(url);
    } catch(err) {
      return fail(new Error('WebSocket创建失败：' + err.message));
    }

    // 60秒总超时
    const wsTimeout = setTimeout(() => {
      try { ws.close(); } catch(_) {}
      done(segments.join(''));   // 超时返回已识别部分
    }, 60000);

    ws.onopen = () => {
      const FRAME  = 1280;
      let offset   = 0;
      let frameIdx = 0;

      const sendNext = () => {
        if (settled) return;

        if (offset >= pcmArray.length) {
          // 发送结束帧
          try {
            ws.send(JSON.stringify({
              data: { status: 2, format: 'audio/L16;rate=16000', encoding: 'raw', audio: '' }
            }));
          } catch(_) {}
          return;
        }

        const chunk = pcmArray.slice(offset, offset + FRAME);
        const b64   = btoa(String.fromCharCode(...chunk));
        const isFirst = frameIdx === 0;

        const frame = {
          ...(isFirst ? { common: { app_id: XF_APPID } } : {}),
          ...(isFirst ? { business: {
            language: 'zh_cn',
            domain:   'iat',
            accent:   'mandarin',
            vad_eos:  5000,
            dwa:      'wpgs'
          }} : {}),
          data: {
            status:   isFirst ? 0 : 1,
            format:   'audio/L16;rate=16000',
            encoding: 'raw',
            audio:    b64
          }
        };

        try { ws.send(JSON.stringify(frame)); } catch(_) { return; }
        offset   += FRAME;
        frameIdx += 1;
        setTimeout(sendNext, 40);
      };

      sendNext();
    };

    ws.onmessage = (e) => {
      let msg;
      try { msg = JSON.parse(e.data); } catch(_) { return; }

      if (msg.code !== 0) {
        clearTimeout(wsTimeout);
        try { ws.close(); } catch(_) {}
        fail(new Error(`讯飞错误(${msg.code}): ${msg.message}`));
        return;
      }

      if (msg.data?.result) {
        const result = msg.data.result;

        // 提取本帧文字
        let text = '';
        (result.ws || []).forEach(w => w.cw.forEach(c => { text += c.w; }));

        const pgs = result.pgs;  // 'apd'追加 | 'rpl'替换
        const rg  = result.rg;   // 替换范围 [start,end]（1-based）

        if (pgs === 'rpl' && Array.isArray(rg) && rg.length === 2) {
          // 替换指定范围的分段
          segments.splice(rg[0] - 1, rg[1] - rg[0] + 1, text);
        } else {
          // 追加新分段
          segments.push(text);
        }

        // 实时预览
        document.getElementById('textDisplay').value = segments.join('');
      }

      if (msg.data?.status === 2) {
        clearTimeout(wsTimeout);
        try { ws.close(); } catch(_) {}
        done(segments.join(''));
      }
    };

    ws.onerror = (e) => {
      clearTimeout(wsTimeout);
      fail(new Error('WebSocket连接失败，请检查网络'));
    };

    ws.onclose = () => {
      clearTimeout(wsTimeout);
      // 如果还没settled，说明意外关闭，返回已有结果
      done(segments.join(''));
    };
  });
}

// ===================== 录音控制 =====================
async function toggleRecording() {
  if (isRecording) {
    stopRecording();
  } else {
    await startRecording();
  }
}

async function startRecording() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

    const mime = ['audio/webm;codecs=pcm', 'audio/webm', 'audio/ogg']
                  .find(t => MediaRecorder.isTypeSupported(t)) || '';

    mediaRecorder = new MediaRecorder(stream, mime ? { mimeType: mime } : {});
    audioChunks   = [];

    // 开始录音：立即清空文本框
    document.getElementById('textDisplay').value = '';

    mediaRecorder.ondataavailable = (e) => {
      if (e.data && e.data.size > 0) audioChunks.push(e.data);
    };

    mediaRecorder.onstop = async () => {
      // 停止所有音轨
      stream.getTracks().forEach(t => t.stop());
      setBtnProcessing();

      // 保底定时器：35秒后强制恢复按钮
      clearTimeout(safetyTimer);
      safetyTimer = setTimeout(() => {
        showToast('识别超时，请重试');
        setBtnIdle();
      }, 35000);

      try {
        if (audioChunks.length === 0) {
          throw new Error('未录到音频，请检查麦克风');
        }
        const blob   = new Blob(audioChunks, { type: mime || 'audio/webm' });
        const result = await recognizeAudio(blob);

        // 最终结果覆盖文本框
        document.getElementById('textDisplay').value = result;
        if (!result.trim()) showToast('未识别到内容，请重试');

      } catch(err) {
        console.error('识别错误：', err);
        showToast('识别失败：' + err.message);
        document.getElementById('textDisplay').value = '';
      } finally {
        clearTimeout(safetyTimer);
        setBtnIdle();
      }
    };

    mediaRecorder.onerror = (e) => {
      console.error('MediaRecorder错误：', e);
      showToast('录音出错：' + (e.error?.message || '未知错误'));
      isRecording = false;
      setBtnIdle();
    };

    mediaRecorder.start(100);
    isRecording = true;
    setBtnRecording();

  } catch(err) {
    console.error('启动录音失败：', err);
    if (err.name === 'NotAllowedError') {
      showToast('请允许麦克风权限后重试');
    } else if (err.name === 'NotFoundError') {
      showToast('未找到麦克风设备');
    } else {
      showToast('录音启动失败：' + err.message);
    }
    setBtnIdle();
  }
}

function stopRecording() {
  if (mediaRecorder && isRecording) {
    isRecording = false;
    try { mediaRecorder.stop(); } catch(e) { console.error(e); setBtnIdle(); }
  }
}

// ===================== 图片处理 =====================
function addPhoto() {
  if (currentImages.length >= 3) { showToast('最多添加3张图片'); return; }
  document.getElementById('fileInput').click();
}

function handleFileSelect(e) {
  for (const file of e.target.files) {
    if (currentImages.length >= 3) break;
    const reader = new FileReader();
    reader.onload = ev => { currentImages.push(ev.target.result); renderPhotoGrid(); };
    reader.readAsDataURL(file);
  }
  e.target.value = '';
}

function removePhoto(i) { currentImages.splice(i, 1); renderPhotoGrid(); }

function renderPhotoGrid() {
  const grid = document.getElementById('photoGrid');
  grid.innerHTML = '';
  currentImages.forEach((src, i) => {
    const d = document.createElement('div');
    d.className = 'photo-item';
    d.innerHTML = `<img src="${src}"><button class="remove-btn" onclick="removePhoto(${i})">×</button>`;
    grid.appendChild(d);
  });
  if (currentImages.length < 3) {
    const btn = document.createElement('div');
    btn.className = 'add-photo-btn';
    btn.onclick   = addPhoto;
    btn.innerHTML = '<span>＋</span><span>拍照/选图</span>';
    grid.appendChild(btn);
  }
}

// ===================== 添加行 =====================
function addRow() {
  const text = document.getElementById('textDisplay').value.trim();
  if (!text && currentImages.length === 0) { showToast('请先录音或添加图片'); return; }
  records.push({ text, images: [...currentImages] });
  document.getElementById('textDisplay').value = '';
  currentImages = [];
  renderPhotoGrid();
  renderRecordsList();
  showToast(`已添加第 ${records.length} 行 ✅`);
}

// ===================== 记录列表 =====================
function renderRecordsList() {
  const list = document.getElementById('recordsList');
  document.getElementById('recordCount').textContent = `${records.length} 行`;
  document.getElementById('exportBtn').disabled = records.length === 0;
  if (!records.length) {
    list.innerHTML = '<div class="empty-tip">暂无记录，添加第一行吧～</div>';
    return;
  }
  list.innerHTML = '';
  records.forEach((row, i) => {
    const d = document.createElement('div');
    d.className = 'record-item';
    d.innerHTML = `
      <div class="record-item-header">
        <span class="row-num">第 ${i+1} 行</span>
        <button class="delete-row-btn" onclick="deleteRow(${i})">🗑 删除</button>
      </div>
      <div class="record-text">${row.text || '（无文字）'}</div>
      ${row.images.length ? `<div class="record-images">
        ${row.images.map(s => `<img src="${s}">`).join('')}
      </div>` : ''}`;
    list.appendChild(d);
  });
}

function deleteRow(i) { records.splice(i, 1); renderRecordsList(); showToast('已删除'); }

// ===================== 导出Excel =====================
async function exportExcel() {
  if (!records.length) return;
  showToast('正在生成Excel，请稍候...');
  document.getElementById('exportBtn').disabled = true;
  try {
    const res = await fetch('/generate-excel', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ rows: records })
    });
    if (!res.ok) throw new Error('服务器错误 ' + res.status);
    const blob = await res.blob();
    const a    = document.createElement('a');
    a.href     = URL.createObjectURL(blob);
    a.download = `语音拍照记录_${new Date().toLocaleDateString('zh-CN').replace(/\//g,'')}.xlsx`;
    a.click();
    showToast('导出成功 📊');
  } catch(err) {
    showToast('导出失败：' + err.message);
  } finally {
    document.getElementById('exportBtn').disabled = false;
  }
}

// ===================== Toast =====================
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2500);
}
